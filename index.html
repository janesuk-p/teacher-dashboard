<script>

// ==============================

// CONFIG

// ==============================

const SHEET_ID = "1ec157T1IhxX_9I5PeBCHdY8nWlxo61N8k6OhKU30Xyg";


// คอลัมน์ที่มีลิงก์ (ใส่ชื่อคอลัมน์ที่มี URL) — ระบบจะตรวจอัตโนมัติด้วย

const LINK_KEYWORDS = ["ลิงก์","link","url","ไฟล์","file","https"];


// คอลัมน์ที่ใช้ Filter (ระบบจะตรวจอัตโนมัติ แต่ระบุได้)

const FORCE_FILTER_COLS = ["ประเภท","ผู้เผยแพร่"];

// ==============================


let allData=[], headers=[], filteredData=[], charts={};


// ---- FETCH ----

async function fetchData(){

  const url=`https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:json`;

  try{

    const r=await fetch(url);

    const t=await r.text();

    const j=JSON.parse(t.match(/setResponse\(([\s\S]*)\)/)[1]);

    const tbl=j.table;

    headers=tbl.cols.map(c=>c.label||c.id||"col");

    allData=tbl.rows.map(row=>

      Object.fromEntries(headers.map((h,i)=>[h,row.c[i]?.v??""]))

    );

    set("statusTxt",`✅ โหลดสำเร็จ ${allData.length} รายการ | ${new Date().toLocaleTimeString('th-TH')}`);

    buildFilters(); applyFilters();

  }catch(e){

    set("statusTxt","❌ โหลดไม่สำเร็จ — ตรวจสอบการแชร์ Sheets ให้เป็น Public");

  }

}


// ---- FILTERS ----

function isLinkCol(h){return LINK_KEYWORDS.some(k=>h.toLowerCase().includes(k.toLowerCase()))}


function buildFilters(){

  const filterCols = headers.filter(h=>{

    if(isLinkCol(h)) return false;

    if(FORCE_FILTER_COLS.includes(h)) return true;

    const vals=allData.map(r=>r[h]).filter(v=>v!=="");

    const u=new Set(vals);

    return u.size>1 && u.size<=30;

  });


  const grid=document.getElementById("filterGrid");

  grid.innerHTML="";


  // Search box

  const sd=document.createElement("div"); sd.className="fg";

  sd.innerHTML=`<label>🔍 ค้นหา</label><input type="text" id="searchBox" placeholder="พิมพ์เพื่อค้นหา..." oninput="applyFilters()"/>`;

  grid.appendChild(sd);


  filterCols.slice(0,4).forEach(col=>{

    const vals=[...new Set(allData.map(r=>r[col]).filter(v=>v!==""))].sort();

    const d=document.createElement("div"); d.className="fg";

    d.innerHTML=`<label>${col}</label>

      <select id="f_${esc(col)}" onchange="applyFilters()">

        <option value="">ทั้งหมด</option>

        ${vals.map(v=>`<option value="${v}">${v}</option>`).join("")}

      </select>`;

    grid.appendChild(d);

  });


  const bd=document.createElement("div"); bd.className="fg";

  bd.style.display="flex"; bd.style.alignItems="flex-end";

  bd.innerHTML=`<button class="btn-refresh" onclick="fetchData()">🔄 รีเฟรช</button>`;

  grid.appendChild(bd);

}


function applyFilters(){

  const q=(document.getElementById("searchBox")?.value||"").toLowerCase();


  const filterCols=headers.filter(h=>{

    if(isLinkCol(h)) return false;

    if(FORCE_FILTER_COLS.includes(h)) return true;

    const vals=allData.map(r=>r[h]).filter(v=>v!=="");

    const u=new Set(vals);

    return u.size>1 && u.size<=30;

  });


  filteredData=allData.filter(row=>{

    const matchFilter=filterCols.slice(0,4).every(col=>{

      const sel=document.getElementById(`f_${esc(col)}`);

      return !sel||sel.value===""||String(row[col])===sel.value;

    });

    const matchSearch=!q||headers.some(h=>String(row[h]).toLowerCase().includes(q));

    return matchFilter&&matchSearch;

  });


  updateScorecards(); buildCharts(); buildTable();

}


// ---- SCORECARDS ----

function updateScorecards(){

  set("sc1",allData.length.toLocaleString('th-TH'));

  set("sc2",filteredData.length.toLocaleString('th-TH'));

  const typeCol=headers.find(h=>h.includes("ประเภท")||h.toLowerCase().includes("type"));

  const sc3=typeCol?new Set(filteredData.map(r=>r[typeCol]).filter(v=>v!=="")).size:"–";

  const personCol=headers.find(h=>h.includes("ผู้เผยแพร่")||h.includes("ครู")||h.toLowerCase().includes("name"));

  const sc4=personCol?new Set(filteredData.map(r=>r[personCol]).filter(v=>v!=="")).size:"–";

  set("sc3",sc3); set("sc4",sc4);

}


// ---- CHARTS ----

const COLORS=["#667eea","#764ba2","#f093fb","#4facfe","#43e97b","#fa709a","#fee140","#30cfd0","#f5576c","#fda085","#84fab0"];


function buildCharts(){

  const area=document.getElementById("chartsGrid");

  area.innerHTML="";


  const catCols=headers.filter(h=>{

    if(isLinkCol(h)||h.toLowerCase().includes("เวลา")||h.toLowerCase().includes("timestamp")) return false;

    const vals=filteredData.map(r=>r[h]).filter(v=>v!=="");

    const u=new Set(vals);

    return u.size>1 && u.size<=20;

  });


  catCols.slice(0,4).forEach((col,i)=>{

    const counts={};

    filteredData.forEach(row=>{

      const v=row[col]!==""?String(row[col]):"(ไม่ระบุ)";

      counts[v]=(counts[v]||0)+1;

    });

    const labels=Object.keys(counts), values=Object.values(counts);

    const isDonut=labels.length<=8;


    const card=document.createElement("div"); card.className="card";

    card.innerHTML=`<h3>${isDonut?"🍩":"📊"} ${col}</h3><div class="chart-wrap"><canvas id="ch${i}"></canvas></div>`;

    area.appendChild(card);


    if(charts[i]) charts[i].destroy();

    const ctx=document.getElementById(`ch${i}`).getContext("2d");

    charts[i]=new Chart(ctx,{

      type:isDonut?"doughnut":"bar",

      data:{

        labels,

        datasets:[{

          data:values,

          backgroundColor:COLORS.slice(0,labels.length),

          borderWidth:isDonut?2:0,borderColor:"white",

          borderRadius:isDonut?0:6

        }]

      },

      options:{

        responsive:true,maintainAspectRatio:false,

        plugins:{

          legend:{position:isDonut?"right":"top",labels:{font:{size:11,family:"'Sarabun',sans-serif"},boxWidth:14}},

          tooltip:{callbacks:{label:c=>{

            const tot=c.dataset.data.reduce((a,b)=>a+b,0);

            return ` ${c.label}: ${c.parsed}  (${((c.parsed/tot)*100).toFixed(1)}%)`;

          }}}

        },

        scales:isDonut?{}:{y:{beginAtZero:true,grid:{color:"#f3f4f6"}},x:{grid:{display:false}}}

      }

    });

  });

}


// ---- TABLE ----

function buildTable(){

  const visibleCols=headers.filter(h=>!isLinkCol(h));

  const linkCols=headers.filter(h=>isLinkCol(h));


  document.getElementById("tHead").innerHTML=

    `<tr>${visibleCols.map(h=>`<th>${h}</th>`).join("")}${linkCols.map(h=>`<th>${h}</th>`).join("")}<th>รายละเอียด</th></tr>`;


  const rows=filteredData.slice(0,300);

  document.getElementById("tBody").innerHTML=rows.map((row,i)=>{

    const cells=visibleCols.map(h=>`<td>${row[h]!==""?`<span class="badge">${row[h]}</span>`:'<span style="color:#d1d5db">–</span>'}</td>`).join("");

    const lCells=linkCols.map(h=>{

      const v=String(row[h]||"");

      return `<td class="link-cell">${v&&v.startsWith("http")?`<a href="${v}" target="_blank" onclick="event.stopPropagation()">🔗 เปิดลิงก์</a>`:'–'}</td>`;

    }).join("");

    return `<tr onclick="openModal(${i})">${cells}${lCells}<td style="color:#667eea;font-weight:600;white-space:nowrap">👁 ดูรายละเอียด</td></tr>`;

  }).join("");


  set("rowCount",`(${filteredData.length} รายการ)`);

}


// ---- MODAL ----

function openModal(idx){

  const row=filteredData[idx];

  const typeCol=headers.find(h=>h.includes("ประเภท"));

  const title=typeCol&&row[typeCol]?row[typeCol]:"รายละเอียดข้อมูล";


  document.getElementById("modalTitle").innerHTML=

    `<span class="type-badge">${title}</span>`;


  const html=headers.map(h=>{

    const v=String(row[h]||"");

    if(v==="") return "";

    const isLink=v.startsWith("http");

    const valueHtml=isLink

      ?`<a href="${v}" target="_blank" class="field-link-btn">🔗 เปิด ${h}</a>`

      :`<span>${v}</span>`;

    return `<div class="modal-field">

      <div class="field-label">${h}</div>

      <div class="field-value">${valueHtml}</div>

    </div>`;

  }).join("");


  document.getElementById("modalContent").innerHTML=html;

  document.getElementById("modalOverlay").classList.add("open");

}


function closeModal(){document.getElementById("modalOverlay").classList.remove("open")}

function closeModalOutside(e){if(e.target.id==="modalOverlay")closeModal()}

document.addEventListener("keydown",e=>{if(e.key==="Escape")closeModal()});


// ---- DOWNLOAD ----

function dlCSV(){

  const bom="\uFEFF";

  const rows=[headers,...filteredData.map(r=>headers.map(h=>`"${String(r[h]).replace(/"/g,'""')}"`))]

    .map(r=>r.join(",")).join("\n");

  trigger(new Blob([bom+rows],{type:"text/csv;charset=utf-8;"}),"data.csv");

}

function dlXLSX(){

  const ws=XLSX.utils.aoa_to_sheet([headers,...filteredData.map(r=>headers.map(h=>r[h]))]);

  const wb=XLSX.utils.book_new();

  XLSX.utils.book_append_sheet(wb,ws,"Sheet1");

  XLSX.writeFile(wb,"data.xlsx");

}

function trigger(blob,name){

  const a=Object.assign(document.createElement("a"),{href:URL.createObjectURL(blob),download:name});

  document.body.appendChild(a);a.click();document.body.removeChild(a);

}


// ---- UTILS ----

function set(id,v){const el=document.getElementById(id);if(el)el.textContent=v}

function esc(s){return s.replace(/[^a-zA-Z0-9ก-๙]/g,"_")}


// ---- INIT ----

fetchData();

setInterval(fetchData,5*60*1000);
